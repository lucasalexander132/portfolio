```typescript
export function Footer() {
  const t = useTranslations()
  const { setCursorVariant, resetCursor } = useCursor()
  // ...

  return (
    <footer>
      {/* Brand */}
      <p>{t('footer.copyright')}</p>
      <p>{t('footer.brand_description')}</p>

      {/* Navigation */}
      <h3>{t('footer.nav_title')}</h3>
      {quickLinks.map(({ href, labelKey, cursorKey }) => (
        <a
          ...
          onMouseEnter={() => setCursorVariant('text', t(cursorKey), true)}
        >
          {t(labelKey)}
        </a>
      ))}

      {/* Contact button */}
      <button
        onMouseEnter={() => setCursorVariant('text', t('footer.cursor.contact'), true)}
      >
        {t('nav.contact')}
      </button>

      {/* Social */}
      <h3>{t('footer.connect_title')}</h3>
      {socialLinks.map(({ href, labelKey, cursorKey, icon: Icon }) => (
        <a
          ...
          onMouseEnter={() => setCursorVariant('text', t(cursorKey), true)}
        >
          <Icon />
          {/* Social labels stay as-is (GitHub, LinkedIn, Bluesky are proper nouns) */}
        </a>
      ))}

      {/* Bottom */}
      <p>... {t('footer.all_rights')}</p>
      <p>{t('footer.availability')}</p>
    </footer>
  )
}
```
